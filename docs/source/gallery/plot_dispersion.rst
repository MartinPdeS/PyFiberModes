    ax = figure.append_ax(show_legend=True)

    for mode in [HE11, HE12, HE22]:
        data = []
        for fiber in factory:

            effective_index = fiber.get_dispersion(
                mode=mode,
                wavelength=1550e-9
            )

            data.append(effective_index)

        ax.add_line(

            y=data,
            label=mode,
            line_width=2
        )

    figure.show()
