    ax.set(
        title='Mode fields for vectorial mode if x-direction',
        xlabel='Radius',
        ylabel='Amplitude'
    )

    for mode in [HE11, LP01, LP11]:

        field = Field(
            fiber=fiber,
            mode=mode,
            limit=10e-6,
            n_point=101
        )

        Ex = field.Ex()
        max_abs = abs(max(Ex.max(), Ex.min()))
        ax.pcolormesh(Ex, vmin=-max_abs, vmax=max_abs)


    plt.show()
